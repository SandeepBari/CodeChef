#include <stdio.h>

main()
{
	int testCases= 0, num1= 0, num2= 0;
    scanf("%d",&testCases);
    
	while(testCases--)
	{
	    scanf("%d %d",&num1, &num2);
	    
	    if (num1<num2)
	    {
	        printf("<\n");
	        continue;
	    }
	    
	    if (num1>num2)
	    {
	        printf(">\n");
	        continue;
	    }
	    
	    if (num1==num2)
	    {
	        printf("=\n");
	        continue;
	    }
	}
}
